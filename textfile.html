    <div class="container">
        <h1>üåê IP Address Checker</h1>
        <button id="getIpBtn" onclick="getIP()">Get My IP Address</button>
        <div class="result" id="result">
            <span class="loading">Click the button to fetch your IP</span>
        </div>
        <button id="copyBtn" class="copy-btn" onclick="copyIP()">Copy IP</button>
    </div>

    <script>
        let currentIP = '';

        async function getIP() {
            const btn = document.getElementById('getIpBtn');
            const result = document.getElementById('result');
            const copyBtn = document.getElementById('copyBtn');
            
            btn.disabled = true;
            result.innerHTML = '<span class="loading">Fetching your IP...</span>';
            copyBtn.style.display = 'none';
            
            try {
                const response = await fetch('https://api.ipify.org?format=json');
                const data = await response.json();
                currentIP = data.ip;
                
                result.innerHTML = `
                    <div class="label">Your IP Address:</div>
                    <div class="ip-address">${currentIP}</div>
                `;
                copyBtn.style.display = 'inline-block';
            } catch (error) {
                result.innerHTML = '<span class="error">Failed to fetch IP. Please try again.</span>';
                console.error('Error:', error);
            } finally {
                btn.disabled = false;
            }
        }

        function copyIP() {
            const copyBtn = document.getElementById('copyBtn');
            
            navigator.clipboard.writeText(currentIP).then(() => {
                const originalText = copyBtn.textContent;
                copyBtn.textContent = '‚úì Copied!';
                copyBtn.classList.add('copied');
                
                setTimeout(() => {
                    copyBtn.textContent = originalText;
                    copyBtn.classList.remove('copied');
                }, 2000);
            }).catch(err => {
                console.error('Failed to copy:', err);
            });
        }
    </script>
